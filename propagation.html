from sgraal import SgraalClient
client = SgraalClient("sg_demo_playground")

result = client.preflight(
    memory_state=[...],
    domain="fintech",
    action_type="irreversible"
)

trace = client.propagation_trace(
    agent_id="agent_001",
    memory_state=[...],
    domain="fintech"
)

print(trace["containment"])       # "SUCCESS"
