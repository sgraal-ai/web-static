Integration

write_firewall_check.py

# Write firewall — block before storage
result = client.write_check(
    entry={"content": "...", "source": "external"},
    namespace="payments"
)
if result.firewall_blocked:
    print(result.block_reason)  # SLEEPER_PATTERN_DETECTED
    print(result.threat_score)  # 0.94

# Read scan — detect poisoning
preflight = client.preflight(
    memory_state=[{"id": "mem_001", "source": "external_feed"}],
    action_type="financial",
    domain="fintech"
)
print(preflight.poisoning_suspected)  # True
print(preflight.recommended_action)   # BLOCK
